## 第30课：异或运算常见题型

### 异或运算交换两数的值

```java
public class Code01_SwapExclusiveOr {

	public static void main(String[] args) {
		int a = -2323;
		int b = 10;
		a = a ^ b;
		b = a ^ b;
		a = a ^ b;
		System.out.println(a);
		System.out.println(b);

		int[] arr = { 3, 5 };
		swap(arr, 0, 1);
		System.out.println(arr[0]);
		System.out.println(arr[1]);
		swap(arr, 0, 0);
		System.out.println(arr[0]);
	}

	// 当i!=j，没问题，会完成交换功能
	// 当i==j，会出错
	// 所以知道这种写法即可，并不推荐
	public static void swap(int[] arr, int i, int j) {
		arr[i] = arr[i] ^ arr[j];
		arr[j] = arr[i] ^ arr[j];
		arr[i] = arr[i] ^ arr[j];
	}

}
```

```c++
#include <iostream>
#include <vector>
using namespace std;

void swap(vector<int>& arr, int i, int j) {
    // 当 i != j 时，使用异或交换
    arr[i] = arr[i] ^ arr[j];
    arr[j] = arr[i] ^ arr[j];
    arr[i] = arr[i] ^ arr[j];
}

int main() {
    int a = -2323;
    int b = 10;

    // 使用异或运算交换两个变量的值
    a = a ^ b;
    b = a ^ b;
    a = a ^ b;

    cout << a << endl;
    cout << b << endl;

    // 定义一个数组
    vector<int> arr = {3, 5};

    // 交换数组中两个不同位置的元素
    swap(arr, 0, 1);
    cout << arr[0] << endl;
    cout << arr[1] << endl;

    // 测试当 i == j 时的情况
    swap(arr, 0, 0);
    cout << arr[0] << endl;

    return 0;
}
```
